# ğŸ”¥ Redirect HTTP to HTTPS

# ğŸš€ Reverse proxy and load balancing 
upstream node_backends {
    server 127.0.0.1:3000;
    server 127.0.0.1:3001;
    # You can configure weights or health checks here
    # server 127.0.0.1:3000 weight=2;
}

server {
    listen 443 ssl;
    server_name app.kartikeynarayan.in;

    # ğŸ“œ SSL certificate and key (Cloudflare origin certificates)
    ssl_certificate /etc/nginx/ssl/cloudflare.pem;
    ssl_certificate_key /etc/nginx/ssl/cloudflare.key;

    # ğŸ” Enforce HSTS for security
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # âš¡ Enable GZIP compression
    gzip on;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_vary on;

    # âœ… Reverse Proxy with Load Balancing
    location / {
        proxy_pass http://node_backends;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # âœ… Increase max body size for large requests
    client_max_body_size 100M;

    # âš ï¸ Custom error page for backend failures
    error_page 502 503 504 /50x.html;

    location = /50x.html {
        root /usr/share/nginx/html;
        internal;
    }
}

# ğŸ”— Enable the site configuration by creating a symbolic link
sudo ln -s /etc/nginx/sites-available/reverse-proxy /etc/nginx/sites-enabled/

# ğŸ” Test the NGINX configuration for syntax errors
